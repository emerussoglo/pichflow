WHERE parent_id IN (
    SELECT id FROM factures WHERE user_id = (SELECT id FROM users WHERE email = 'stoicisme69@gmail.com')
    UNION
    SELECT id FROM devis WHERE user_id = (SELECT id FROM users WHERE email = 'stoicisme69@gmail.com')
